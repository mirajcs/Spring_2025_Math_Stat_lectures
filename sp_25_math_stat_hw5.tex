\begin{parts}
    \part[] \(P(X>10)\)

        \(P(X>10)=1-P(X \leq 10)=1- \Phi(0.5)=0.3085\)
    \end{solution}
    \part[] \(P(-20 <X< 15)\)
    \begin{solution}
        \(\Phi(1)-\Phi(-2.5)=\Phi(1)-1+\Phi(2.5)=0.8351\)
    \end{solution}
    \part[] the value of \(x\) such that \(P(X>x)=0.065\)

\end{parts}
